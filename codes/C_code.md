```c
/*
작성자 : Larkka
날짜 : 2019.10.31

입력으로 바둑판의 어떤 상태가 주어졌을 때, 검은색이 이겼는지,
흰색이 이겼는지 또는 아직 승부가 결정되지 않았는지를 판단하는 프로그램을 작성하시오.
단, 검은색과 흰색이 동시에 이기거나 검은색 또는 흰색이 두 군데 이상에서 동시에 이기는 경우는 입력으로 들어오지 않는다.

8줄에 각 줄마다 8개의 숫자로 표현되는데, 검은 바둑알은 1, 흰 바둑알은 2, 알이 놓이지 않는 자리는 0으로 표시되며, 숫자는 한 칸씩 띄어서 표시된다.

첫줄에 검은색이 이겼을 경우에는 1을, 흰색이 이겼을 경우에는 2를, 아직 승부가 결정되지 않았을 경우에는 0을 출력한다.
검은색 또는 흰색이 이겼을 경우에는 둘째 줄에 연속된 다섯 개의 바둑알 중에서 가장 왼쪽에 있는 바둑알
(연속된 다섯 개의 바둑알이 세로로 놓인 경우, 그 중 가장 위에 있는 것)의 가로줄 번호와, 세로줄 번호를 순서대로 출력한다.
*/


#include <stdio.h>
int main()
{
	int a[8][8];//바둑판 저장 변수(이차원 배열 8 x 8)
	int x, y;//첫번째 바둑알의 행, 열 저장 변수 

	printf("=====오목 승패 판단 프로그램=====\n\n검은바둑돌 : %d\n흰 바둑돌 : %d\n\n\n", 1, 2);

	//바둑판 상태 입력 8 x 8
	printf("   1 2 3 4 5 6 7 8\n   ---------------\n");
	for (int i = 0; i < 8; i++)
	{
		printf("%c |", 'a' + i);
		for (int j = 0; j < 8; j++)
		{
			scanf(" %d", &a[i][j]);
		}
	}

	//승패 판정 알고리즘

	//가로 오목
	for (int i = 0; i < 8; i++)
	{
		for (int j = 0; j < 8; j++)
		{
			if (a[i][j] == a[i + 1][j] && a[i][j] == a[i + 2][j] && a[i][j] == a[i + 3][j] && a[i][j] == a[i + 4][j])
			{
				if (a[i][j] == 1)
				{
					printf("\n검은색 승리");
					x = i;
					y = j + 1;

				}
				else if (a[i][j] == 2)
				{
					printf("\n흰색 승리");
					x = i;
					y = j + 1;
				}
			}
		}
	}

	//세로 오목
	for (int i = 0; i < 8; i++)
	{
		for (int j = 0; j < 8; j++)
		{
			if (a[i][j] == a[i][j + 1] && a[i][j] == a[i][j + 2] && a[i][j] == a[i][j + 3] && a[i][j] == a[i][j + 4])
			{
				if (a[i][j] == 1)
				{
					printf("\n검은색 승리");
					x = i;
					y = j + 1;
				}
				else if (a[i][j] == 2)
				{
					printf("\n흰색 승리");
					x = i;
					y = j + 1;
				}
			}
		}
	}

	//대각선 오목
	for (int i = 0; i < 8; i++)
	{
		for (int j = 0; j < 8; j++)
		{
			//오른쪽 아래 방향 대각선
			if (a[i][j] == a[i + 1][j + 1] && a[i][j] == a[i + 2][j + 2] && a[i][j] == a[i + 3][j + 3] && a[i][j] == a[i + 4][j + 4])
			{
				if (a[i][j] == 1)
				{
					printf("\n검은색 승리");
					x = i;
					y = j + 1;
				}
				else if (a[i][j] == 2)
				{
					printf("\n흰색 승리");
					x = i;
					y = j + 1;
				}
			}

			//왼쪽 아래 방향 대각선
			else if (a[i][j] == a[i - 1][j + 1] && a[i][j] == a[i - 2][j + 2] && a[i][j] == a[i - 3][j + 3] && a[i][j] == a[i - 4][j + 4])
			{
				if (a[i][j] == 1)
				{
					printf("\n검은색 승리");
					x = i;
					y = j + 1;
				}
				else if (a[i][j] == 2)
				{
					printf("\n흰색 승리");
					x = i;
					y = j + 1;
				}
			}
		}
	}

	//결과 출력
	printf("\n첫번째 바둑알은 %c행 %d열에 있습니다.", 'a' + x, y);


	return 0;
}
```
